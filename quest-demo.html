üéØ Unity Integration

Add streaming to your Unity VR game

using SubstreamSDK;

public class VRStreamController : MonoBehaviour
{
    private Substream.LiveHandle liveHandle;
    
    async void Start()
    {
        // Initialize in demo mode
        await Substream.Init("demo");
    }
    
    public async void StartStreaming()
    {
        var config = new Substream.LiveConfig
        {
            Width = 1920,
            Height = 1080,
            Fps = 30,
            BitrateKbps = 5000
        };
        
        liveHandle = await Substream.StartLive(config);
        
        // Listen for status updates
        liveHandle.OnStatusChanged += (status) => {
            Debug.Log($"Stream status: {status}");
        };
    }
    
    public async void StopStreaming()
    {
        await liveHandle?.Stop();
    }
}
