using SubstreamSDK;
using UnityEngine;

public class StreamController : MonoBehaviour
{
    private Substream.LiveHandle liveStream;
    
    async void Start()
    {
        // Initialize SDK (use real credentials in production)
        await Substream.Init("demo");
    }
    
    public async void StartStreaming()
    {
        // Configure stream quality
        var config = new Substream.LiveConfig
        {
            Width = 1920,
            Height = 1080,
            Fps = 30,
            BitrateKbps = 5000
        };
        
        // Start streaming - SDK handles permissions automatically
        liveStream = await Substream.StartLive(config);
        
        // Get viewer link to share
        string viewerUrl = Substream.GetViewerLink();
        Debug.Log($"Share this link: {viewerUrl}");
    }
    
    public async void StopStreaming()
    {
        await liveStream?.Stop();
    }
}
// Android/Quest Native Implementation
class VRStreamActivity : AppCompatActivity() {
    
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        
        // Initialize Substream SDK
        SubstreamNative.init(this, SubstreamConfig(
            baseUrl = "demo",  // Use real URL in production
            whipPublishUrl = "wss://your-server/whip"
        ))
    }
    
    fun startStreaming() {
        // SDK handles MediaProjection permission request
        SubstreamNative.startLive(
            width = 1920,
            height = 1080,
            fps = 30,
            bitrateKbps = 5000
        )
        
        // Listen for status updates
        SubstreamNative.setStatusCallback { status ->
            when (status) {
                StreamStatus.CONNECTED -> showViewerLink()
                StreamStatus.ERROR -> showError()
            }
        }
    }
}
